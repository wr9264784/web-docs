index.js：

// 获取 左侧学科列表 和 右侧课程表 的父级元素
// 使用事件委托的方式监听父级元素下面所有子元素的拖拽事件（这样方便一点，也可以监听每一个子元素的拖拽事件）
const container = document.querySelector('.school-timetable-wrap')

// 存储拖拽元素
let source = null

// 监听拖拽开始的事件
// 开始拖动元素时触发
container.ondragstart = e => {
  // 通过 e.target 可以知道被拖拽元素是哪个
  console.log('dragstart', e)

  source = e.target

  // 设置拖拽时，鼠标的状态
  e.dataTransfer.effectAllowed = e.target.dataset.effect
}

// 当被拖动的对象在目标容器范围内拖动时触发此事件
// 这个事件触发的比较频繁，因为拖拽这个元素的时候会经过很多其他元素。
// 像这里的 div、table、th、td、tr 等。
// 只要被拖拽元素在某个元素上面就会不停地触发。
container.ondragover = e => {
  // 通过 e.target 就可以知道当前被拖拽元素在哪个元素的上面
  // console.log('dragover', e)

  // 拖拽事件需要阻止浏览器默认事件，否则目标元素事件 `drop` 事件，用户看不到效果
  e.preventDefault()
}

// 当被鼠标拖动的对象进入目标容器时触发此事件，这个事件只触发一次
container.ondragenter = e => {
  // 通过 e.target 就可以知道当前被拖拽元素在哪个元素的上面
  // console.log('dragenter', e)

  // 当被拖拽元素移入某个目标元素时，改变目标元素的背景颜色
  // 注意：
  // 此时我们拖拽元素到目标元素的时候，会发现右侧的表格全部都改变了背景颜色，
  // 这是因为这个 dragenter 事件，不仅会在子元素中触发，在父元素上也会触发。
  // 所以，当被拖拽元素进入目标元素的时候，会触发目标元素的父元的 dragenter 事件。
  // 也就是说父元素先改变背景颜色，然后再改变子元素的背景颜色。
  //
  // 所以，我们这里要仔细思考一下。
  // 我们到底要改变哪个元素的背景颜色，是不是就是取决于当前被拖拽元素能拖拽到哪个地方。
  // 那么，我们怎么知道哪些元素是能接受拖拽的？同样的，需要用到元素标签的自定义属性。
  // // e.target.classList.add('drop-over')

  /************************************************************************************* */

  // 应该按照以下方法实现：

  // 每次进入目标元素前，先清除上一次已经设置了背景颜色的元素
  clearDropStyle()

  // // 获取进入的目标元素
  // const dropNode = e.target

  // // 只有目标元素的自定义 drop 属性值 和 被拖拽的元素的鼠标状态值相同，
  // // 目标元素才能接收该拖拽元素，目标元素才能改变背景颜色。
  // if (dropNode.dataset.drop === e.dataTransfer.effectAllowed) {
  //   e.target.classList.add('drop-over')
  // }

  /************************************************************************************* */

  // 完成了上述代码后，我们发现还是会有问题：
  // 当我们拖拽 学科元素 到 目标元素 时，目标元素 有可能也是 学科元素，
  // 因为有可能是上一次已经完成了拖拽的事件，上一次的 学科元素 已经被 放在了课程表的 表格中了。
  // 这个时候我们就不能给 这个目标元素（学科元素）设置背景颜色了，而是这个目标元素的父级元素。
  // 所以，这里改变不一定是目标元素，也有可能是目标元素的祖先元素。
  // 只要目标元素的祖先元素满足一个条件，也就是设置了 data-drop 属性的祖先元素就行。

  // 获取设置了 drop 属性的元素
  const dropNode = getDropNode(e.target)

  // 只有目标元素的自定义 drop 属性值 和 被拖拽的元素的鼠标状态值相同，
  // 目标元素才能接收该拖拽元素，目标元素才能改变背景颜色。
  if (dropNode && dropNode.dataset.drop === e.dataTransfer.effectAllowed) {
    dropNode.classList.add('drop-over')
  }
}

// 在一个拖动过程中，释放鼠标键到目标容器时触发此事件
container.ondrop = e => {
  // 通过 e.target 就可以知道当前被拖拽元素在哪个元素的上面
  console.log('ondrop', e)

  // 先清除背景颜色
  clearDropStyle()

  // 获取设置了 drop 属性的元素
  const dropNode = getDropNode(e.target)

  // 判断目标元素能否接收这个拖拽的元素
  if (dropNode && dropNode.dataset.drop === e.dataTransfer.effectAllowed) {
    // 如果拖拽元素，鼠标状态是 copy，说明是 学科 拖拽到 课程表，
    // 复制该拖拽元素，添加到 目标元素 中
    if (dropNode.dataset.drop === 'copy') {
      dropNode.innerHTML = ''

      // 深度复制被拖拽元素
      const cloneNode = source.cloneNode(true)

      // 修改 拖拽元素 放到 目标元素 后，该元素的鼠标状态
      cloneNode.dataset.effect = 'move'

      // 添加到目标元素中
      dropNode.appendChild(cloneNode)
    }

    // 否则，就是 课程表 中的 学科 拖拽到 学科列表
    else {
      // 删除课程表中的学科元素
      source.remove()
    }
  }
}


// 清除背景颜色
function clearDropStyle() {
  document.querySelectorAll('.drop-over').forEach(node => {
    node.classList.remove('drop-over')
  })
}

// 获取设置了 drop 的元素
function getDropNode(node) {
  while (node) {
    // 如果发现了设置 drop 属性的元素，返回该元素
    if (node.dataset && node.dataset.drop) return node

    // 如果没有找到，获取该元素的父级元素，
    // 继续 while 循环判断父级元素是否有 drop 属性
    // 直到找不到为止。
    node = node.parentNode
  }
}
