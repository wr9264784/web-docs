简单的拖拽

拖拽效果如下所示：

index.html：

<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta http-equiv="X-UA-Compatible" content="IE=edge">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>拖拽</title>
  <link rel="stylesheet" href="./index.css">
</head>
<body>
  <div class="box-wrap">
    <div id="box1" class="box">box1 - 可以拖拽</div>
    <div id="box2" class="box">box2 - 不可拖拽</div>
  </div>

  <script src="./index.js"></script>
</body>
</html>

index.js：

/**
 * 拖拽流程
 * 1. 当鼠标在被拖拽元素上被按下时，开始拖拽 onmousedown
 * 2. 当鼠标移动时，被拖拽元素随鼠标移动 onmousemove
 * 3. 当鼠标松开时，被拖拽元素固定在当前位置 onmouseup
 */
const box1 = document.getElementById('box1')

// 为 box1 绑定鼠标按下事件
box1.onmousedown = (ev) => {
  // 阻止浏览器默认行为
  ev.preventDefault()

  console.log('鼠标按下，开始拖拽')

  // 当我们每次点击 box1 盒子的时候，不管我们点击盒子的哪个位置，
  // 当我们开始移动鼠标时，盒子会发生 “偏移”，默认以盒子左上角为基准，
  // 移动到我们鼠标的位置，我们需要优化这个问题。
  // 
  // box1 的偏移量：
  // 鼠标.clientX（鼠标在可视区域的水平位置） - 元素.offsetLeft（元素原来的位置距最近的祖先定位元素可视区域左侧的位置）
  // 鼠标.clientY - 元素.offsetTop
  // 
  // 解决方案就是，鼠标按下的时候，计算盒子偏移的距离
  // 然后在鼠标移动时，让盒子回到没有发生偏移前的位置。
  const oldLeft = ev.clientX - box1.offsetLeft
  const oldTop = ev.clientY - box1.offsetTop


  // monmousemove 不能给 box1 绑定，
  // 为 document 绑定一个 onmousemove 事件，
  // 让 box1 元素跟随鼠标移动
  document.onmousemove = (e) => {
    e.preventDefault()

    // 获取鼠标的坐标
    // const left = e.clientX
    // const top  = e.clientY

    // 优化偏移问题，减去盒子的偏移量，让盒子不发生偏移
    const left = e.clientX - oldLeft
    const top  = e.clientY - oldTop

    // 修改 box1 的位置
    box1.style.left = left + 'px'
    box1.style.top = top + 'px'
  }

  // 为 document 元素绑定一个鼠标松开事件
  document.onmouseup = () => {
    // 取消 document 的 onmouseup 事件，
    // 因为不取消，点击盒子以外的位置，松开鼠标时，仍然会触发 document.onmouseup 事件。
    document.onmouseup = null

    // 当鼠标松开时，被拖拽元素固定在当前位置
    // 取消 document 的 onmousemove 事件
    document.onmousemove = null
  }

  // 取消浏览器默认行为
  // return false
}
