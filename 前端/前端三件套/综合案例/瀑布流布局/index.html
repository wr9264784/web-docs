index.js：

const imgList = []
const imgWidth = 220 // 每张图片的固定宽度

const box = document.querySelector('.container')

createImgs()

// 这里还不能一开始就直接调用设置图片位置的方法，
// 因为调用该方法后，虽然页面上的图片元素已经通过 createImgs 方法创建了，
// 但是加载图片是需要一段时间的，我们可以在浏览器的网络选项卡中查看请求，
// 我们可以看到当前的 index.js 代码已经运行了，但是图片还没有请求加载出来。
// 因此它得到的图片高度永远是 0。
// setImgPosition()

// 我们应该在 window.onload，也就是页面资源全部加载完成后才调用。
// 但是，这样又有一个问题：
// 当我们的网络较慢的时候，浏览器会慢慢地把图片加载出来，
// 但是没有一次性把所有的图片加载出来，
// 因此这个 setImgPosition 还没有运行，
// 所以一开始看上去会是一个非常丑的页面，图片都叠加在一起。
// 那么我们应该怎么处理呢？
// 其实很简单，只要有一张图片加载出来了，就应该调用一次 setImgPosition 方法。
// 即：我们应该在 createImgs 方法中的 for 循环里面，监听图片的加载完成事件，
// 只要某张图片加载完成了，就调用 setImgPosition 方法。
// window.onload = setImgPosition

let timer = null

// 监听窗口大小变化，重新设置图片位置，
// 并用防抖处理过于频繁的情况
window.onresize = function() {
  if (timer) clearTimeout(timer)
  timer = setTimeout(setImgPosition, 300)
}

// 加入图片元素
function createImgs() {
  // 循环 50 次，每次生成 10 张图片
  for (let n = 0; n <= 10; n++) {
    for (let i = 1; i <= 10; i++) {
      // 生成图片的 src 路径
      const src = '../../../assets/images/' + i + '.jpeg'
      const img = document.createElement('img')

      // 实际项目中根据你们的图片路径，动态生成 src 即可
      img.src = src
  
      // 当图片加载完成后，调用设置图片位置方法
      img.onload = setImgPosition
  
      // 把图片添加到容器中
      box.appendChild(img)
    }
  }
}

// 设置图片的宽度
function setStyles() {
  // 获取样式变量
  // document.documentElement 获取 html 元素
  // getComputedStyle() 方法用于获取指定元素的 CSS 样式
  // getPropertyValue() 获取一个 Dom 节点上的 CSS 变量
  getComputedStyle(document.documentElement).getPropertyValue('--img-width'); 

  // 设置样式变量
  document.documentElement.style.setProperty('--img-width', '220px');
}

/**
 * 计算容器盒子一共有多少列，以及每一列之间的间隙
 */
function calc() {
  const boxWidth = box.clientWidth // 获取容器的宽度
  const columns = Math.floor(boxWidth / imgWidth) // 计算有多少列
  
  // 计算每一列之间的间隙
  const gapNum = columns + 1 // 间隙数量
  const space = boxWidth - columns * imgWidth // 计算剩余的空间
  const gap = space / gapNum // 每个间隙的空间

  return {
    gap,
    columns
  }
}

/**
 * 设置每张图片的位置
 */
function setImgPosition() {
  // 调用计算函数，获取 列数 和 间隙的大小
  const boxInfo = calc()

  // 创建一个数组，数组的长度为容器的列数
  // 该数组的作用是：记录容器每一列的高度，
  // 并把最小高度的那一列的高度值作为下一张要插入的图片的 top 值。
  const nextTops = new Array(boxInfo.columns)

  // 将数组的每一项都填充为 0，
  // 也就是说初始化时，每一列的高度都为 0
  nextTops.fill(0)

  // 设置图片位置
  for (let i = 0; i < box.children.length; i++) {
    const img = box.children[i]

    // 找到 nextTops 中的最小值作为当前图片的纵坐标
    // 换句话说就是：找到数组中高度最小的那一列，并在该列中插入图片，图片的 top 值就是该列的高度
    const minTop = Math.min.apply(null, nextTops)

    img.style.top = minTop + 'px' // 设置图片位置

    // 重新设置数组这一项的下一个 top 值
    const index = nextTops.indexOf(minTop) // 得到当前使用的是第几列的 top 值
    nextTops[index] += img.height + boxInfo.gap // 更新当前使用的那一列的高度

    const left = (index + 1) * boxInfo.gap + index * imgWidth

    img.style.left = left + 'px'
  }

  // 获取高度最高的那一列的高度
  const max = Math.max.apply(null, nextTops)
  
  // 设置容器的高度
  box.style.height = max + 'px'
}
