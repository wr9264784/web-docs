export.js：

/**
 * 导出 excel
 * @param { object[] } data 需要导出的数据
 * @param { String } sheetName 工作表的名称
 * @param { String } exportName 导出的 excel 文件名称
 */
function exportToExcel(sheetName, exportName, data, headers) {
  const workbook = XLSX.utils.book_new() // 创建一个工作簿
  const worksheet = XLSX.utils.json_to_sheet(data) // 创建一个工作表
  console.dir(worksheet);
  // 将工作表添加到工作簿中， 新工作表将被称为传入的 "sheetName"
  // 默认情况下，json_to_sheet 创建一个带有标题行的工作表，标头来自 JS 对象键。
  XLSX.utils.book_append_sheet(workbook, worksheet, sheetName)

  // 如果有传入的自定义的表头名称
  if (headers && headers.length > 0) {
    // XLSX.utils.sheet_add_aoa 可以将文本值写入从单元格 A1 开始的现有工作表
    XLSX.utils.sheet_add_aoa(worksheet, [headers], { origin: "A1" });
  }

  // writeFile 尝试将 wb 写入指定 filename 的本地文件。
  // 在浏览器中，它会尝试提示用户下载文件。 
  // 在 NodeJS 中，它将写入本地目录。
  // 第三个参数是 写作选项，具体参数需参考官网：https://xlsx.nodejs.cn/docs/api/write-options/
  XLSX.writeFile(workbook, `${exportName}.xlsx`, { compression: true })
}

import.js：

/**
 * 导入 excel 数据
 * @param { ArrayBuffer } file 要读取的文件 buffer 缓冲区
 * @returns 生成的 HTML 字符串
 */
function importExcel(file) {
  // 解析文件
  // read 尝试解析 file 并返回 工作簿对象
  // 第二个参数是 options 对象，options 的 type 决定了 data 的解释方式。 对于字符串数据，默认解释是 Base64。
  const workbook = XLSX.read(file)

  // 检查工作表数据
  // Sheets 是一个对象，其键是工作表名称，其值是工作表对象。
  // SheetNames 属性是工作簿中工作表名称的有序列表。 
  // 例如，通过索引 SheetNames 并使用名称索引 Sheets 来拉取第一个工作表：
  const firstSheet = workbook.Sheets[workbook.SheetNames[0]]

  // sheet_to_html 从工作表对象生成 HTML 表
  const table = XLSX.utils.sheet_to_html(firstSheet)
  
  return table
}

index.js：

const exportUrl = 'https://sheetjs.com/data/executive.json' // 导出数据的 url
const importUrl = 'https://sheetjs.com/data/PortfolioSummary.xls' // 导入数据的 url
let exportData = []

// 获取要导出的数据
async function getData() {
  let res = await (await fetch(exportUrl)).json()
  exportData = res
}
getData()

const exportBtn = document.querySelector('.export-btn')
const ipmortBtn = document.querySelector('.import-btn')

// 导出按钮点击
exportBtn.onclick = function() {
  console.log('开始导出')
  const start = Date.now()
  const data = exportData.map(item => {
    return {
      id: item.id.bioguide + item.id.govtrack + item.id.icpsr_prez,
      name: item.name.first + item.name.last,
      gender: item.bio.gender,
      birthday: item.bio.birthday
    }
  })
  exportToExcel('sheet1', 'excel', data, ['ID', '姓名', '性别', '出生年月'])
  const end = Date.now()
  console.log(`导出结束，耗时 ${end - start} ms`);
}

// 导入按钮点击
ipmortBtn.onclick = async function() {
  const file = await (await fetch(importUrl)).arrayBuffer()
  const tableContent = importExcel(file)
  const table = document.querySelector('.table')
  table.innerHTML = tableContent
}
