获取视频流播放视频

后端实现

后端通过 express 编写一段简单的返回 视频流 的接口：

// 引入Express框架
const express = require("express");
// 引入文件系统模块
const fs = require("fs");
const path = require("path");

// 创建一个Express应用实例
const app = express();

// 设置服务器监听的端口
const port = 3000;

// 定义一个路由，处理对视频文件的请求
app.get("/api/getVideo", (req, res) => {
  // 视频文件的路径（请替换为实际的视频文件路径）
  const videoPath = path.resolve(__dirname, "./media/video.mp4");

  // 使用 fs.statSync 同步方法获取视频文件的状态信息
  const stat = fs.statSync(videoPath);

  // 获取文件大小（以字节为单位）
  const fileSize = stat.size;

  // 设置响应头信息
  const head = {
    // 指明发送给接受方的消息主体的大小。Content-Length 如果存在并且有效地话，
    // 则必须和消息内容的传输长度完全一致。否则就会导致异常 (特别地, HTTP1.0 中这个字段可有可无)
    "Content-Length": fileSize,
    "Content-Type": "video/mp4", // 设置内容类型为视频格式
    "Access-Control-Allow-Origin": "*",
    "Access-Control-Allow-Headers": "*",
    "Access-Control-Allow-Methods": "*"
  };

  // 设置HTTP状态码为200（成功）
  res.writeHead(200, head);

  // 创建一个只读流并管道到响应对象中，开始发送数据
  fs.createReadStream(videoPath).pipe(res);
});

// 启动服务器，监听指定的端口
app.listen(port, () => {
  console.log(`Server is running on port ${port}`);
});

前端实现 - 方式1

在前端中，可以直接通过 video 标签的 src 属性设置请求视频流的 api 接口：

<!DOCTYPE html>
<html lang="en">

<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>视频流</title>
</head>

<body>
  <!-- 直接通过给 video 标签的 src 设置获取视频流的 api 请求，会获取视频流进行播放 -->
  <video src="http://localhost:3000/api/getVideo" controls></video>
</body>

</html>

前端实现 - 方式2

在前端中，还可以通过点击某个按钮发送请求，获取视频流并下载并设置 video 的 src 进行播放：

<!DOCTYPE html>
<html lang="en">

<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>视频流</title>
</head>

<body>
  <button>点击获取视频</button>
  <video class="video" controls width="250"></video>

  <script src="https://unpkg.com/axios/dist/axios.min.js"></script>
  <script type="module">
    const btn = document.querySelector('button')
    btn.onclick = async () => {
      try {
        const res = await axios({
          url: 'http://localhost:3000/api/getVideo',
          method: 'get',
          responseType: "blob" // 重点，需要设置响应数据类型
        })
        const url = URL.createObjectURL(res.data)
        const video = document.querySelector('video')
        video.src = url

        // 下载文件
        const link = document.createElement('a')
        link.style.display = 'none'
        link.href = url
        link.setAttribute('download', 'video.mp4')
        document.body.appendChild(link)
        link.click()
        document.body.removeChild(link)
      } catch (error) {
        console.log(error);
      }
    }
  </script>
</body>

</html>
