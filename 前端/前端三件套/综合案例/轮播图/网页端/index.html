index.js：

// 所有涉及到的 dom 元素
const doms = {
  carouselList: document.querySelector('.carousel-list'),
  arrowLeft: document.querySelector('.carousel-arrow-left'),
  arrowRight: document.querySelector('.carousel-arrow-right'),
  indicator: document.querySelectorAll('.indicator span')
}

const count = doms.indicator.length // 轮播图的数量
let curIndex = 0 // 当前显示的轮播的索引

/**
 * 跳转至第几个索引的轮播图
 * @param { Number } index 轮播图的索引
 */
function moveTo(index) {
  doms.carouselList.style.transform = `translateX(-${index}00%)` // 跳转第几张轮播
  doms.carouselList.style.transition = '.5s' // 添加过渡效果

  // 去掉指示器的选中效果
  const active = document.querySelector('.indicator span.active')
  active.classList.remove('active')

  // 添加选中的指示器
  doms.indicator[index].classList.add('active')

  curIndex = index // 记录当前的索引
}

// 指示器注册点击事件
doms.indicator.forEach((item, i) => {
  item.onclick = function() {
    moveTo(index)
  }
})

/**
 * 初始化函数，为了实现无缝轮播，
 * 需要在轮播图列表的 第一张 前面 插入最后一张轮播图，
 * 在轮播图列表的 最后一张 后面 插入第一张轮播图。
 */
function init() {
  // 复制轮播图列表的第一个轮播图元素
  const first = doms.carouselList.firstElementChild.cloneNode(true)

  // 复制轮播图列表的最后一个轮播图元素
  const last = doms.carouselList.lastElementChild.cloneNode(true)

  // 但是这样会有一个问题，就是我们把最后一张元素插入第一张元素之前，
  // 那么，默认就会显示最后一张图的元素了。
  // 所以，我们需要让插入后的第一张轮播元素发生偏移。
  last.style.marginLeft = '-100%'

  doms.carouselList.appendChild(first) // 把第一张轮播图插入末尾

  // 把最后一个轮播图元素插入到第一张的前面
  doms.carouselList.insertBefore(last, doms.carouselList.firstElementChild)
}

init() // 一开始就调用初始化函数

/**
 * 点击左侧按钮，实现无缝轮播
 */
function moveLeft() {
  // 如果已经是第一张了，进行无缝轮播
  if (curIndex === 0) {
    /**
     * 无缝轮播要怎么实现呢？
     * 1. 我们可以先快速地把这个界面给它跳转到最后一个元素（插入的那个最后一个元素），
     * 因为索引为 0 时的轮播图和插入到末尾的最后一个元素是一样的，当我们快速跳转时，
     * 用户是感知不到的，没有任何变化的。
     * 
     * 2. 然后在最后一张轮播的基础上，往前移动一张轮播图，就可以实现无缝轮播。
     */
    doms.carouselList.style.transform = `translateX(-${count}00%)` // 移动的距离就是轮播图的数量 * 100% 的宽度

    // 但是这里还有问题，如果我们先点击了指示器，跳转到其他索引的轮播图，
    // 然后再回到第一张轮播图，再点击左箭头，会发现有跳转动画，这是因为
    // 在点击指示器的时候设置了过渡效果，所以这里需要清除掉这个过渡效果。
    doms.carouselList.style.transition = 'none'

    // moveTo(count - 1) // 移动到倒数第二张（也就是插入前的最后一张）

    // 但是，这里又有问题了，因为当前函数中设置了 doms.carouselList.style.transition = 'none'，
    // 而 moveTo() 里又设置了 doms.carouselList.style.transition = '.5s'。
    // 所以，当前函数中的  doms.carouselList.style.transition = 'none' 相当于白设置了。
    // 不产生效果，这是由于浏览器的渲染原理导致的。因为当前的 js 代码还没有执行完成，页面是不会渲染的。
    // 所以，当设置了 doms.carouselList.style.transition = 'none' 又马上设置 doms.carouselList.style.transition = '.5s'，
    // 最终会执行 doms.carouselList.style.transition = '.5s' 的效果。
    // 
    // 所以，可以等待浏览器渲染完成后，再执行 moveTo()
    requestAnimationFrame(() => {
      moveTo(count - 1) // 移动到倒数第二张（也就是插入前的最后一张）
    })
  } else {
    // 跳转到当前轮播的上一张
    moveTo(curIndex - 1)
  }
}

/**
 * 点击右侧按钮，实现无缝轮播
 */
function moveRight() {
  // 如果已经是最后一张了，进行无缝轮播
  if (curIndex === count - 1) {
    doms.carouselList.style.transform = `translateX(100%)`
    doms.carouselList.style.transition = 'none'

    requestAnimationFrame(() => {
      moveTo(0) // 移动到第二张（也就是插入前的第一张）
    })
  } else {
    // 跳转到当前轮播的下一张
    moveTo(curIndex + 1)
  }
}

doms.arrowLeft.onclick = moveLeft
doms.arrowRight.onclick = moveRight
