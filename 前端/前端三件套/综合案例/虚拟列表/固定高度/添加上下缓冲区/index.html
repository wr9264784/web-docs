固定高度优化 - 添加上下缓冲区

index.html：

<!DOCTYPE html>
<html lang="en">

<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>虚拟列表（优化：添加上下缓冲区）</title>
  <link rel="stylesheet" href="./index.css">
</head>

<body>
  <div id="container" onscroll="handleScroll()">
    <!-- 模拟滚动条 -->
    <!-- 用于撑开 container 容器，使 container 容器可以滚动 -->
    <!-- scroll-bar 盒子的高度：列表的总数量 * 每个列表项的高度 -->
    <div class="scroll-bar"></div>

    <!-- 用于渲染真实显示数量的列表 -->
    <!-- scroll-list 盒子的高度：真实渲染的列表项数量 * 每个列表项的高度 -->
    <!-- scroll-list 盒子的高度 必然小于 container 的高度 -->
    <div class="scroll-list"></div>
  </div>

  <script src="./index.js"></script>
</body>

</html>

index.js：

const list = [] // 原始列表
let visualList = [] // 截取后的列表
const total = 999 // 列表总数
const size = 40 // 每个列表项的高度
const marginBottom = 12 // 每个列表项的样式中的 margin-bottom
const listItemTotalHeight = size + marginBottom // 每个列表项 + 列表项 marginBottom 的总高度
let num = 0 // 可见的列表项数量
let startIndex = 0 // 列表截取开始的索引
let endIndex = 0 // 列表截取结束的索引
let prevNum = 0 // 当前可视区域的上面预留几个，相当于在可视区域上面预留一个可视区域
let nextNum = 0 // 当前可视区域的下面预留几个，相当于在可视区域下面预留一个可视区域

/**
 * 获取所有列表数据
 */
function getAllList() {
  for (let i = 0; i < total; i++) {
    list.push({
      id: i,
      value: i + 1
    })
  }
}

/**
 * 获取可视区域的列表项数量
 */
function setViewNum() {
  const height = document.documentElement.clientHeight
  num = Math.ceil(height / listItemTotalHeight) // 设置可视区域的列表项数量
  endIndex = num - 1
}

/**
 * 获取当前可视区域上面和下面的预留区域列表项的数量，
 * 在 vue 项目中，应该用计算属性来分别返回 prevNum 和 nextNum。
 */
function getPrevNumAndNextNum() {
  prevNum = Math.min(startIndex, num) // 上面预留几个
  nextNum = Math.min(num, total - endIndex) // 下面预留几个
}

/**
 * 设置 scrool-bar 的高度
 */
function setHeight() {
  const scrollBar = document.querySelector('.scroll-bar')

  // 盒子的高度为：列表总数 * 每个列表项的高度
  scrollBar.style.height = total * listItemTotalHeight + 'px'
}

/**
 * 获取可视数量的列表
 * 如果是在 Vue 框架中，获取截取的列表：应该通过计算属性的方式动态计算
 */
function getvisualList() {
  const start = startIndex - prevNum
  const end = endIndex + nextNum
  // 截取需要显示对应数量的列表
  visualList = list.slice(start, end + 1)

  const ele = document.querySelector('.scroll-list')
  ele.innerHTML = ''

  visualList.forEach((item, i) => {
    const div = document.createElement('div')
    div.classList.add('scroll-list-item')
    div.innerHTML = `ID: ${item.id}, VALUE: ${item.value}`
    ele.appendChild(div)
  })
}

/**
 * 滚动事件
 */
function handleScroll() {
  const ele = document.querySelector('#container')
  const scrollTop = ele.scrollTop // 获取滚动的高度
  calcIndex(scrollTop) // 计算开始和结束索引
  getPrevNumAndNextNum() // 获取预留数量
  translate() // 偏移位置
  getvisualList() // 重新获取截取后的列表
}

/**
 * 计算列表截取开始的索引和结束的索引，
 */
function calcIndex(scrollTop) {
  // 应该先计算出当前滚上去几个了，当前就应该从第几个开始显示。
  // 获取滚动过去了几个元素，从而获取到当前应该从第几个元素开始渲染
  // (滚动的距离 scrollTop) 除以 (每个列表项的高度：size + marginBottom) 等于 (滚动的个数)。
  // 这里用 Math.floor 是因为：如果我们滚动了某个元素的一半高度，但是这个元素
  // 还没有完全滚动过去，我们仍然是要显示该元素，所以要向下取整。
  // 例如：滚动过去了 2.5 个元素，即第二个元素滚动了一半，我们仍然是要显示第二个元素。
  startIndex = Math.floor(scrollTop / listItemTotalHeight)

  endIndex = startIndex + num // 获取截取结束的索引
}

/**
 * scroll-list 可视区域的偏移位置。
 */
function translate() {
  // 获取可视区盒子
  const ele = document.querySelector('.scroll-list')

  // 调整可视区域位置
  // scroll-list 是相对于 container 盒子进行定位的，
  // 当 container 盒子网上滚动过去了 x 个元素后，
  // scroll-list 盒子就应该向下偏移对应的距离，从而保持 scroll-list 盒子相对于 container 盒子的可视区域不变。
  // 如果有预留区域，需要减去
  const offset = startIndex * listItemTotalHeight - prevNum * listItemTotalHeight
  ele.style.transform = `translateY(${offset}px)`
}
