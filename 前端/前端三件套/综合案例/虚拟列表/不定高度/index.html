index.js：

const list = [] // 原始列表
let visualList = [] // 截取后的列表
const total = 100 // 列表总数
const size = 100 // 每个列表项的 “预估高度”，如果预估计值给的太小，有可能在滚动的时候，当前元素只滚动了一半，就开始计算下一个元素的索引
const marginBottom = 12 // 每个列表项的样式中的 margin-bottom
const listItemTotalHeight = size + marginBottom // 每个列表项 + 列表项 marginBottom 的总高度
let num = 0 // 可见的列表项数量
let startIndex = 0 // 列表截取开始的索引
let endIndex = 0 // 列表截取结束的索引
let prevNum = 0 // 当前可视区域的上面预留几个，相当于在可视区域上面预留一个可视区域
let nextNum = 0 // 当前可视区域的下面预留几个，相当于在可视区域下面预留一个可视区域
let positions = [] // 记录元素的位置信息

/**
 * 获取所有列表数据
 */
function getAllList() {
  for (let i = 0; i < total; i++) {
    const randomNum = Math.floor(Math.random() * 100) + 20 // 生成随机整数 20 - 100
    list.push({
      id: i,
      // value: Mock.Random.sentence(5, 50) // 通过 mock.js 生成随机字符串
      value: getRandomString(randomNum)
    })
  }
}


/**
 * 随机生成字符串
 * @param len 指定生成字符串长度
 */
function getRandomString(len){
  let _charStr = 'abacdefghjklmnopqrstuvwxyzABCDEFGHJKLMNOPQRSTUVWXYZ0123456789'
  let min = 0
  let max = _charStr.length - 1
  let _str = ''; // 定义随机字符串 变量
  len = len || 15; // 判断是否指定长度，否则默认长度为 15

  // 循环生成字符串
  for(var i = 0, index; i < len; i++){
    index = (function(randomIndexFunc, i) {         
      return randomIndexFunc(min, max, i, randomIndexFunc);
    })(function(min, max, i, _self){
      let indexTemp = Math.floor(Math.random()*(max-min+1)+min),
          numStart = _charStr.length - 10;
      if(i==0&&indexTemp >=numStart){
          indexTemp = _self(min, max, i, _self);
      }
      return indexTemp ;
    }, i)
    _str += _charStr[index];
  }
  return _str;
}

/**
 * 获取可视区域的列表项数量
 */
function setViewNum() {
  const height = document.documentElement.clientHeight
  num = Math.ceil(height / listItemTotalHeight) // 设置可视区域的列表项数量
  endIndex = num - 1
}

/**
 * 获取当前可视区域上面和下面的预留区域列表项的数量，
 * 在 vue 项目中，应该用计算属性来分别返回 prevNum 和 nextNum。
 */
function getPrevNumAndNextNum() {
  prevNum = Math.min(startIndex, num) // 上面预留几个
  nextNum = Math.min(num, total - endIndex) // 下面预留几个
}

/**
 * 设置 scrool-bar 的高度
 */
function setHeight() {
  const scrollBar = document.querySelector('.scroll-bar')

  // 盒子的高度为：列表总数 * 每个列表项的高度
  scrollBar.style.height = total * listItemTotalHeight + 'px'
}

/**
 * 获取可视数量的列表
 * 如果是在 Vue 框架中，获取截取的列表：应该通过计算属性的方式动态计算
 */
function getvisualList() {
  const start = startIndex - prevNum
  const end = endIndex + nextNum
  // 截取需要显示对应数量的列表
  visualList = list.slice(start, end + 1)

  const ele = document.querySelector('.scroll-list')
  ele.innerHTML = ''

  visualList.forEach((item, i) => {
    const div = document.createElement('div')
    div.classList.add('scroll-list-item')
    div.innerHTML = `${item.value}`
    div.dataset.id = `${item.id}` // 设置定义 data- 属性，方便后续 update 函数使用
    ele.appendChild(div)
  })
}

/**
 * 如果加载完毕，需要缓存每一项的高度、top 值、还有 bottom
 */
function cacheListItem() {
  // 1. 先记录好预估计的列表项数值，等滚动的时候、渲染页面时获取
  //    真实 dom 的高度来更新缓存的内容。
  // 2. 再重新计算滚动条的高度
  positions = list.map((item, index) => ({
    height: size,
    top: index * size,
    bottom: (index + 1) * size
  }))
}

/**
 * 滚动事件
 */
function handleScroll() {
  const ele = document.querySelector('#container')
  const scrollTop = ele.scrollTop // 获取滚动的高度

  // 需要使用二分查找法，找到对应的的记录
  startIndex = getStartIndex(scrollTop)
  endIndex = startIndex + num
  getPrevNumAndNextNum() // 获取预留数量
  update()
  translate() // 偏移位置
  getvisualList() // 重新获取截取后的列表
}

/**
 * 使用二分查找法，找到开始的索引
 * @param { Number } value 当前滚动的需要找到的值
 */
function getStartIndex(value) {
  let start = 0 // 开始的索引
  let end = positions.length - 1 // 结束的索引
  let temp = null  // 临时中间值

  while( start <= end) {
    // 找到中间的索引
    let middleIndex = parseInt((start + end) / 2)

    // 找到中间的索引，就可以找到中间的 positions 元素
    // 找到当前的中间的那个元素的结尾点
    let middleValue = positions[middleIndex].bottom

    // 如果直接找到了，就返回当前的下一个元素，
    // 因为找到的是当前元素的底部，下一个元素才是要显示的元素。
    if (middleValue === value) {
      return middleIndex + 1
    }

    // 如果没有找到，拿当前要找的元素和中间的元素进行对比，
    // 看要找的元素是在左边还是有右边。
    // 如果要查找的元素在右边
    else if (middleValue < value) {
      start = middleIndex + 1
    }

    // 如果要查找的元素在左边
    else if (middleValue > value) {
      // 不断缩小范围
      if (temp === null || temp > middleIndex) temp = middleIndex
      end = middleIndex - 1
    }
  }

  return temp
}

/**
 * 核心方法：
 * 根据当前显示的内容，更新缓存中的 height、bottom、top，
 * 最终更新滚动条的高度
 */
function update() {
  // 获取真实渲染的节点
  const eles = document.querySelectorAll('.scroll-list-item')

  if (eles && eles.length > 0) {
    eles.forEach(item => {
      let { height } = item.getBoundingClientRect() // 当前真实的高度

      // 把旧的高度更新为当前真实的高度
      const id = item.dataset.id - 0
      const oldEleHeight = positions[id].height

      // 计算当前的高度是否和之前的高度有变化
      const diff = oldEleHeight - height

      // 将后续的所有元素都要向下偏移
      if (diff) {
        positions[id].height = height

        // 高度变化了，top 值不会改变还是原来的值
        //（bottom 值）减去（高度变化的值）等于（bottom 变化后的值）
        positions[id].bottom = positions[id].bottom - diff
        
        // 有一个变化了高度，后面的所有元素都需要变化
        // 这里从当前元素的后一个元素开始
        for (let i = id + 1; i < positions.length; i++) {
          //（后一个元素的 top） 就应该等于（上一个元素的 bottom + margin-bttom 的样式（如果有设置的话））
          positions[i].top = positions[i - 1].bottom + marginBottom

          // 后一个元素的 bottom 也是相同的变化公式
          positions[i].bottom = positions[i].bottom - diff
        }
      }
    })

    // 更新滚动条的高度
    const scrollBar = document.querySelector('.scroll-bar')
    scrollBar.style.height = positions[positions.length - 1].bottom + marginBottom + 'px'
  }
}

/**
 * scroll-list 可视区域的偏移位置。
 */
function translate() {
  // 获取可视区盒子
  const ele = document.querySelector('.scroll-list')

  // 设置偏移量
  // startIndex - prevNum 相减后有可能没有值
  const offset = positions[startIndex - prevNum] ? 
    positions[startIndex - prevNum].top
    :
    0
  ele.style.transform = `translateY(${offset}px)`
}

/**
 * 节流函数
 * @param { Function } func 要执行的函数
 * @param { Number } delay 延迟时间 ms
 * @returns 
 */
function throttle(func, delay) {
  let timer = null

  return function() {
    const context = this
    const args = arguments

    if (timer) return
  
    timer = setInterval(() => {
      func.apply(context, args)
      clearInterval(timer)
      timer = null
    }, delay)
  }
}

const container = document.querySelector('#container')
container.addEventListener('scroll', handleScroll)
container.addEventListener('scroll', throttle(handleScroll, 300))
