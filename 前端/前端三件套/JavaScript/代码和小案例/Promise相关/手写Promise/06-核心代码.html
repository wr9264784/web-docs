// 优化1. 记录 Promise 的三种状态
const PENDING = 'pending'
const FULFILLED = 'fulfilled'
const REJECTED = 'rejected'

/**
 * 运行一个微队列任务，把传递的函数放到微队列中
 * @param { Function } callback 需要添加到微队列执行的函数
 */
function runMicroTask(callback) {
  // node 环境
  if (typeof process !== 'undefined' && typeof process.nextTick === 'function') {
    // 把回调函数放到微队列里面
    process.nextTick(callback)
  }

  // 浏览器环境
  else if (MutationObserver) {
    // 手动创建一个 div 元素
    const div = document.createElement('div')

    // 创建一个 MutationObserver 观察器实例，并传递一个回调
    const observer = new MutationObserver(callback)

    // 通过观察器的 observe 方法，监听 div 元素
    observer.observe(div, {
      childList: true // 观察该元素内容的变化
    })

    // 手动去改变这个 div 元素的内部内容，
    // 相当于手动触发了 MutationObserver() 传入的 callback
    div.innerHTML = '1'
  }

  // 其他情况
  else {
    setTimeout(callback, 0)
  }
}

/**
 * 判断是否为 Promise 函数
 */
function isPromiseLike(value) {
  /**
   * 这里不能简单地通过 instanceof 来判断 Promise，因为 Promise 的范畴是很广的。
   * 在 ES6 里面它使用了一个构造函数来帮你构造建一个 Promise。
   * 
   * ES6 之前也是有 Promise 的，它的时间顺序是这样的，先出来的是一套规范，叫做 Promise/A+ 规范，
   * 这是一个社区标准。它告诉我们如何来处理异步场景。那时候 ES6 还没出来。
   * 它的意思就是说你只要满足这套标准，你就是一个 Promise，并且你们之间还可以进行互操作。
   * 
   * 而我们的 ES6 里面的那个构造函数，无非就是去实现了这个标准而已。
   * 
   * 因此我们这里要判断它是不是一个 Promise，就意味着要判断的目标并不是说要判断它是否为 ES6 的
   * Promise 构造函数，而是它满不满足 Promise/A+ 规范。
   * 
   * 那么该如何判断是否满足该规范呢？
   * 这个 Promise 可以是一个对象，也可以是一个函数，只要它带有一个属性 then 方法，并且这个属性
   * then 是一个方法。
   */
  // return value instanceof Promise

  // 这里 !! 的作用是让最终返回的结果是一个 Boolean 类型，
  // 因为 value 的值是 unfined 或者 0 等为 false 的值时，
  // 会直接返回 value 本身，这不是我们想要的。我们想要的是 Boolean。
  return !!value && typeof value.then === 'function'
}

class MyPromise {
  /**
   * 创建一个 Promise 实例对象
   * @param { Function } executor 执行器，实例化这个类时，这个函数会立即执行
   */
  constructor(executor) {
    this._state = PENDING // Promise 的初始化状态
    this._value = undefined // Promise 的数据（保存的是成功或失败状态的数据）
    this._handlers = [] // 创建一个队列数组

    /**
     * 立即执行该执行器并捕获 executor 执行过程的报错
     */
    try {
      executor(this._resolve.bind(this), this._reject.bind(this))
    } catch (error) {
      // Promise 执行过程中，如果报错，手动调用 reject() 函数。
      this._reject(error)
    }
  }

  /**
   * 改变状态的方法
   * @param { String } newState 新的状态
   * @param { any } newValue 新的状态得到的数据
   */
  _changeState(newState, newValue) {
    // 在 Promise 中，只要状态一旦发生变化，当前状态就不能再次被修改
    if (this._state !== PENDING) return

    this._state = newState
    this._value = newValue

    // 当改变了状态的时候，无论是成功还是失败，都得运行队列里的函数
    this._runHandlers()
  }

  /**
   * 标记当前任务完成的函数
   * 无论是完成还是失败，resolve 和 reject 方法实际上要做的是改变 Promise 的状态。
   * @param { any } data 表示任务完成时的相关数据
   */
  _resolve(data) {
    // 改变状态和数据
    this._changeState(FULFILLED, data)
  }

  /**
   * 标记当前任务失败的函数
   * 无论是完成还是失败，resolve 和 reject 方法实际上要做的是改变 Promise 的状态。
   * @param { any } reason 表示任务失败时的相关数据
   */
  _reject(reason) {
    // 改变状态和数据
    this._changeState(REJECTED, reason)
  }

  /**
   * 添加函数到队列中的方法
   * @param { Function } executor then 需要执行的回调函数
   * @param { String } state Promise 的状态，不同的状态下，then 需要执行不同的回调函数
   * @param { Function } resolve 让 then 函数返回的 Promise 调用 resolve，改变 Pormise 状态
   * @param { Function } reject 让 then 函数返回的 Promise 调用 reject，改变 Pormise 状态
   */
  _pushHandler(executor, state, resolve, reject) {
    // 往队列中添加一个对象，对象中包含了 then 的回调 和 执行回调时的 Promise 状态
    this._handlers.push({
      executor, // then 方法要执行的函数
      state, // Promise 为什么状态时，执行函数 executor
      resolve, // 成功时要执行的改变状态函数
      reject // 失败时要执行的改变状态函数
    })
  }

  /**
   * 根据实际情况，执行队列里的每一个函数
   * 状态发生变化 和 then 方法中都需要调用。
   */
  _runHandlers() {
    if (this._state === PENDING) return

    // 循环获取队列里的每一个函数，每次都拿第一项
    while(this._handlers[0]) {
      // 每次执行第一项
      this._runOneHandler(this._handlers[0])

      // 每判断处理一次队列的第一项函数，就把它删掉
      this._handlers.shift()
    }
  }

  /**
   * 执行队列中的其中一个函数（核心代码）
   * @param { Object } handler 要处理的函数对象
   */
  _runOneHandler({ executor, state, resolve, reject }) {
    // 把队列中的某一个函数，放到微队列中执行
    runMicroTask(() => {
      // 如果当前 Promise 的状态 不等于 handler 所需要执行时的状态
      // 不做任何处理
      if (this._state !== state) return

      /**
       * 如果当前 Promise 的状态 等于 handler 所需要执行时的状态，
       * 但是在调用 then() 的时候没有传递对应的回调，也就是 hanlder 为 undefined。
       * 获取传递了参数，但是该参数不是一个函数，这些都是无效的传递。
       * 
       * 在这种情况下，需要进行 “状态穿透”。什么意思呢？
       * 如果说我们的 Promise 状态发生变化了，但是在 then 方法中没有传递对应的处理，
       * 或者说传递的是无效的处理，那么 then 方法返回的新 Promise 的状态和当前一致，
       * 并且新的 Promise 得到的数据也和当前的一致。
       */ 
      if (this._state === state && typeof executor !== 'function') {
        // 判断当前状态，改变 then() 返回的 Promise 的状态和数据
        this._state === FULFILLED
          ? resolve(this._value)
          : reject(this._value)

        return
      }

      /**
       * 剩下的就是正常的情况，也就是传递的是正常的函数，并且状态也发生了变化。
       * 
       * 这种情况下，当我们状态和 handler 里的状态一致时，就要执行对应状态的函数，
       * 并且 then() 返回的新 Promise 的状态取决于 then() 接收的回调的执行结果。
       * 
       * 如果 then() 接收的函数在执行过程中没有报错，那么这个回调就是成功的，回调
       * 的数据就是通过函数内部 return 返回的结果。
       * 
       * 如果执行过程中报错了，那么它就是失败的，它的数据就是报错的原因。
       */
      try {
        /**
         * 执行 then 函数中接收的回调
         * 
         * ！！注意：！！
         * 如果是通过 handler.executor() 调用，那么会导致 executor 函数内部的
         * this 指向 handler，这是不合适的。
         * 
         * 如果我们通过在 _runOneHandler() 接收参数时，就把 handler 进行解构。
         * 这样，直接调用 executor() 时，this 就不会发生变化。
         * 
         * 并且，调用 executor() 时，是需要传参的，因为 then 函数接收的回调不管
         * 是 onFulfilled 还是 onRejected 处理函数都有一个参数。
         * 
         * 这个参数是什么呢？其实就是当前 Promise 改变状态后的数据。
         * 
         * 并且，executor 函数执行成功后，需要 return 成功后的数据给新的 Promise。
         * 所以，执行成功后，需要拿到返回的结果。
         */
        const res = executor(this._value)

        /**
         * 如果是 Promise
         * 
         * ！！注意：！！
         * 如果拿到的结果又是一个 Promise，那么我们 then 返回的新的 Promise 对象的
         * 状态和数据要和 res 的状态和数据要完全保持一致。
         */
        if (isPromiseLike(res)) {
          // 如果返回的是 Promise，那么它一定有 then 方法，
          // 那么如果说返回的这个 Promise 执行成功了，就运行一下 then 函数要返回的 Promise 的 resolve 方法。
          // 如果说返回的这个 Promise 执行失败了，就运行一下 then 函数要返回的 Promise 的 reject 方法。
          // 这样就能保证获取到的 Promise 和 then 返回要返回的 Promise 状态和数据保持一致。
          res.then(resolve, reject)
        }

        else {
          // 如果调用过程成功，改变 Promise 的状态，并且拿到 executor 的返回结果，
          // 作为新的 Promise 的状态改变后的数据
          resolve(res)
        }
      } catch (error) {
        // 如果调用过程中报错，改变 Promise 的状态，拿到错误的原因，
        // 作为新的 Promise 的状态改变后的数据
        reject(error)
      }
    })
  }

  /**
   * Promise A+ 规范的 then 函数
   * @param { Funtion } onFulfilled 成功后的回调
   * @param { Function } onRejected 失败后的回调
   * @returns 返回一个新的 Promise
   */
  then(onFulfilled, onRejected) {
    // then 函数内一定是返回一个新的 Promise
    return new MyPromise((resolve, reject) => {
      // 把 then 成功和失败的回调都添加到队列中
      this._pushHandler(onFulfilled, FULFILLED, resolve, reject) // 表示这个函数只有成功才执行
      this._pushHandler(onRejected, REJECTED, resolve, reject) // 表示这个函数只有失败才执行

      // 执行队列的函数
      this._runHandlers()
    })
  }
}

// 测试我们的 Promise
const p = new MyPromise((resolve, reject) => {
  setTimeout(() => {
    resolve('pppp')
  }, 500)
})

const _p = p.then(res => {
  // 如果函数执行成功，是需要返回一个数据给新的 Promise 的
  // return 123

  // 如果在回调中又返回一个 Promise 对象，这种情况下，
  // _p 对象和 这里返回的 Promise 对象的状态和数据要保持一致。
  return new MyPromise((resolve, reject) => {
    // resolve('aaa')
    reject('bbb')
  })
})

  console.log(p);
  console.log(_p);
}, 1500)
