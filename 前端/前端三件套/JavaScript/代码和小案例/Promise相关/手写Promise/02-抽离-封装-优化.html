抽离、封装、优化 Promise

// 优化1. 记录 Promise 的三种状态
const PENDING = 'pending'
const FULFILLED = 'fulfilled'
const REJECTED = 'rejected'

class MyPromise {
  /**
   * 创建一个 Promise 实例对象
   * @param { Function } executor 执行器，实例化这个类时，这个函数会立即执行
   */
  constructor(executor) {
    this._state = PENDING // Promise 的初始化状态
    this._value = undefined // Promise 的数据（保存的是成功或失败状态的数据）

    /**
     * 优化4. 捕获 executor 执行过程的报错
     * 如果我们的 Promise 在执行过程中报错了，比方说我在 Promise 的回调中手动
     * 抛出一个错误 throw 123，我们应该需要处理这个错误，把我们的 Promise 状态
     * 自动改为 rejected。
     * 
     * 我们应该怎么处理这件事情呢？我们只需要在调用 executor() 执行器的时候，
     * 进行错误的捕获即可。如果有错误，直接手动调用 reject()。
     */
    try {
      executor(this._resolve.bind(this), this._reject.bind(this)) // 立即执行该执行器
    } catch (error) {
      // Promise 执行过程中，如果报错，手动调用 reject() 函数。
      this._reject(error)
    }
  }

  /**
   * 优化2. 抽离改变状态的方法
   * 因为不管是 resolve 还是 reject 都有在改变状态和记录改变后的数据
   * 所以，可以这把部分的代码抽离出来进行封装。
   * @param { String } newState 新的状态
   * @param { any } newValue 新的状态得到的数据
   */
  _changeState(newState, newValue) {
    // 优化3. 在 Promise 中，只要状态一旦发生变化，当前状态就不能再次被修改
    if (this._state !== PENDING) { // 说明状态已经被改了
      return
    }

    this._state = newState
    this._value = newValue
  }

  /**
   * 标记当前任务完成的函数
   * 无论是完成还是失败，resolve 和 reject 方法实际上要做的是改变 Promise 的状态。
   * @param { any } data 表示任务完成时的相关数据
   */
  _resolve(data) {
    // 改变状态和数据
    this._changeState(FULFILLED, data)
  }

  /**
   * 标记当前任务失败的函数
   * 无论是完成还是失败，resolve 和 reject 方法实际上要做的是改变 Promise 的状态。
   * @param { any } reason 表示任务失败时的相关数据
   */
  _reject(reason) {
    // 改变状态和数据
    this._changeState(REJECTED, reason)
  }
}

// 测试我们的 Promise
const p = new MyPromise((resolve, reject) => {
  // resolve(123)
  // reject(456)

  throw 123
})

console.log(p);
