数字转中文

实现目标

首先看下效果：

当我们点击转换的时候，可以把对应的数字转换成对应的中文，并且显示正确的读法。

代码实现

index.html：

<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>数字转中文</title>
</head>
<body>
  <button>转换</button>

  <div>
    <span>原数字：</span>
    <span class="number">000000001234</span>
  </div>

  <div>
    <span>转换后：</span>
    <span class="chineseNum"></span>
  </div>

  <script src="./index.js"></script>
</body>
</html>

index.js：

/**
 * 数字转换中文
 * @param { Number } num 万亿元以下的正整数
 * 
 * 按照我们人的自身习惯和读法，我们一般都是把数字每四位进行分割，
 * 因为每四位的读法是一致的，都是 个十百千。
 * 
 * 例如这里的 45674567，分割成 4567,4567 分割后这两部分的读法一致。
 * 唯一不同的就是前一部分的单位需要加上一个 万 做单位。
 * 
 * 所以，首先要做的事情就是每四位进行分割。
 */
function toChineseNumber(num) {
  // 这里的使用正则表达式的 前瞻 "?=" 去匹配 "\d{4}" 每四位数字的前面的位置。
  // 并且在每四位数字前面的位置添加逗号。
  const numstr = num.toString().replace(/(?=(\d{4})+$)/g, ',')

  console.log('分割后的数字：', numstr)

  // filter(Boolean) 过滤数组中为空的项
  const numArr = numstr.split(',').filter(Boolean)
  console.log('转换成数组：', numArr);

  let result = ''

  for (let i = 0; i < numArr.length; i++) {
    const part = numArr[i] // 拿到每一项，也就是每四位的值

    // 对每一项分别进行分别的转换
    const cn = transform(part)

    // 对转换后的每一项带上大单位
    let bigUnit = bigUnits[numArr.length - 1 - i]

    // 当 四位 数都是 0 的时候，不需要带大单位
    if (cn === charts[0]) bigUnit = ''

    result += cn + bigUnit

    // 把末尾的零去掉
    result = handleZero(result)
  }

  console.log(result);
  return result
}

/**
 * 辅助函数，给一个四位的整数，返回它对应的中文形式
 * @param { String } 四位的正整数
 * @returns 返回它对应的中文形式
 */
function transform(n) {
  // 如果全部都是零的情况，需要单独处理，返回一个零
  if (n === '0000') return charts[0]

  let result = ''

  // 遍历这个数字字符串，每遍历一次就对单个字符进行转换
  for (let i = 0; i < n.length; i++) {
    // 拿到每一项的中文字符，并转换为数字，
    // 再通过数字拿到映射列表中对应的中文。
    const cn = charts[n[i] - 0]

    /**
     * 取单位时，到底要取哪个单位呢？那就要看这个数字的第几位的数字了。
     * 
     * 比方说：1234 下表为 0 的数字是 1，那么它取的单位是 千，
     * 那么一定是这样吗？那不一定。
     * 
     * 如果说这个数字不是 四位 的，而是 三位 的，比如：123，
     * 那么下标为 0 的数字，那么它取的单位是 百。
     * 
     * 所以，我们应该怎么来建立这个关联呢？
     * 
     * 仔细观察会发现 第几位 数字，也就是 for 循环里的 i 和
     * units 数组的下标有一个什么样的关联呢？
     * 就是数字的长度 - 1 - i
     */
    let unit = units[n.length - 1 - i]

    // 如果某一个数字转换后是中文零的情况，就不需要带单位了
    // 把当前数字对应的单位清空。
    if (cn === charts[0]) unit = ''

    result += cn + unit

    // 处理存在有连续的零的情况
    result = handleZero(result)
  }

  /**
   * 当我们传入的数字是 '1234'，打印的结果为 一千二百三十四。
   * 
   * 1. 
   * 这个时候就没问题了吗？其实不是的，比方说这里传入的 n 为 1204
   * 那么打印的结果是 一千二百零十四，这种说法明显是有问题的。
   * 
   * 所以，还需要判断一下数字为 0 的情况，数字为 0 就不需要带单位了。
   * 
   * 2. 
   * 再改动了某个数字为 0 的情况后，还有没有问题呢？其实还有的。
   * 当我们传入的数字 n 为 1004 的时候，打印的结果为 一千零零四，
   * 这种说法也是错误的。
   * 
   * 也就是最终我们需要把这个字符串中连续的零给它去掉。可以写一个辅助方法：
   * handleZero()，把连续的零转换成只有一个零。
   * 
   * 3. 
   * 此时还没完，如果连续的 0 出现在末尾，比如：1200，那么打印的结果是：
   * 一千二百零。所以，还需要把末尾的 0 全部都去掉。
   * 
   * 还有一种情况，如果 0 出现在 四位数 的开头呢？比如：0120，那么打印
   * 结果为：零一百二十。
   * 
   * 4. 
   * 这种说法合不合理呢？其实是合理的，因为这 四位 是整个数字的一部分。
   * 比方说这个 0120 它一定是出现在整个数字的中间或者末尾的，比方说：
   * 1234,0120,1234 或者 1234,0120，那么不管它在哪个位置，0120 开头
   * 的这个零一定是会读到的。所以可以保留。
   * 
   * 但是，如果全部都是 0 呢？比如：0000，那么它就没东西了，打印为空。
   * 因为我们在第 2、3 点已经做了处理连续的零变成一个零，然后末尾的零
   * 又被去掉了，那么这种情况对于 1234,0000,1234 来说好像又不合理了。
   * 因为 1234,0000,1234 读作 一千二百三十四亿零一千二百三十四，中间的
   * 零是需要读到的。
   * 
   * 所以连续的 4 为 0，需要单独处理。
   */
  return result
}

/**
 * 处理字符串中连续存在的字符零，
 * 把连续的零转换成只有一个零。
 */
function handleZero(str) {
  // 如果字符串中的零出现了 2 次以上，变成一个
  // return str.replace(/零{2,}/g, '零')

  // 此时还没完，如果连续的 0 出现在末尾，比如：1200，那么打印的结果是：
  // 一千二百零。
  // 所以，还需要把末尾的 0 全部都去掉。
  return str.replace(/零{2,}/g, '零').replace(/零+$/g, '')
}

// 数字对应的汉字映射列表
// 可以发现，数字索引和对应索引的中文是一一对应的。
// 例如：索引 0 对应的中文是 零。
const charts = [
  '零', '一', '二', '三', '四',
  '五', '六', '七', '八', '九'
]

// 定义四位数字的单位
// 因为个位在汉字里不进行任何表示，所以给一个空字符串
const units = ['', '十', '百', '千']

// 定义一个大单位
// 比方说 1234,0000,1234，最右侧的四位不需要带大单位
// 中间的 4 位带上大单位 万，最左侧的 四位 带上大单位 亿。
const bigUnits = ['', '万', '亿']

const btn = document.querySelector('button')
btn.onclick = function() {
  const numDom = document.querySelector('.number')
  const chnDom = document.querySelector('.chineseNum')
  const number = numDom.innerHTML
  const chineseNum = toChineseNumber(number)
  chnDom.innerHTML = chineseNum
}
