Canvas 实现放刮刮乐效果

技术点：图像合成。

具体流程：

  1. 我们先通过绘制一个矩形区域作为待刮的区域，
  2. 当鼠标按下的时候记录当前点位，作为 moveTo() 的起点位置。
  3. 然后再鼠标按下的过程中进行移动时，不断地 lineTo() 移动到对应的鼠标位置形成线条。
  4. 并且矩形区域和线条重叠的部分用 图像合成 直的destination-out 模式效果，把重叠的部分镂空。

具体操作逻辑：

// index.js

const canvas = document.getElementById('myCanvas')
const ctx = canvas.getContext('2d')
canvas.width = 300
canvas.height = 100

let isClick = false

// 绘制一个矩形作为待刮的区域
ctx.fillStyle = '#c0bfba'
ctx.fillRect(0, 0, canvas.width, canvas.height)

console.log(canvas.width, canvas.height);

// 绘线条的样式，用作刮动时的线条
ctx.lineCap = 'round'
ctx.lineJoin = 'round'
ctx.lineWidth = 24

canvas.onmousedown = function (e) {
  const x = e.offsetX
  const y = e.offsetY
  ctx.beginPath()

  // 设置绘制的起点为当前点击的位置
  ctx.moveTo(x, y)
  isClick = true
}

canvas.onmousemove = function (e) {
  if (!isClick) return

  const x = e.offsetX
  const y = e.offsetY

  // 使用 destination-out 合成模式，将线条和矩形重叠的区域清除
  ctx.globalCompositeOperation = 'destination-out';
  ctx.lineTo(x, y)
  ctx.stroke()
}

canvas.onmouseup = function (e) {
  isClick = false
}
