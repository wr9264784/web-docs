Canvas 实现放大镜

我们在逛一些商城网站的时候，应该都有注意到，在商品页面中会有一个放大的效果，可以查看商品放大后的细节，如下图所示：

其实这种效果除了用传统的 js 和 css 来实现外，还可以用 Canvas 来实现。

技术点：离屏渲染。

将一个 Canvas 对象绘制到另一个 Canvas 对象上（离开用户可视范围内进行渲染）。

我们可以创建两个 canvas 图片，一个是 放大的图，是一个 原图，如上图中的左边和中间的图片。

当我们鼠标移动到 原图 的某个位置的时候，会在 放大的图 中找到对应的坐标位置，然后将相应的位置裁剪。裁剪后再把裁剪的图片放到 原图 中鼠标点击的位置。

之所以说用到了 离屏渲染 技术，是因为这个 放大的图 对于用户来说是不可见的。

具体代码如下：

index.html 页面结构：

<!DOCTYPE html>
<html lang="en">

<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Canvas 放大镜</title>
  <style>
    * {
      margin: 0;
      padding: 0;
    }
  </style>
</head>

<body>
  <!-- 展示原图的 canvas -->
  <canvas id="myCanvas"></canvas>

  <!-- 用于放大原图的 cavnas，默认不可见 -->
  <canvas id="offCanvas" style="display: none;"></canvas>

  <script src='./index.js'></script>
</body>

</html>

index.js 具体操作逻辑：

// 真正展示的 Cavans
const canvas = document.getElementById('myCanvas')
const ctx = canvas.getContext('2d')
const width = 1080
const height = 1439

// 放图图的 Canvas，用户不可见
const offCanvas = document.getElementById('offCanvas')
const offCtx = offCanvas.getContext('2d')

canvas.width = width / 2
canvas.height = height / 2

offCanvas.width = width
offCanvas.height = height

const scale = 2

const image = new Image()
let isMouseDown = false

image.src = './assets/goddess.jpeg'
image.onload = function () {
  ctx.drawImage(image, 0, 0, canvas.width, canvas.height)
  offCtx.drawImage(image, 0, 0, offCanvas.width, offCanvas.height)
}

// 监听鼠标按下的事件
canvas.onmousedown = function (e) {
  e.preventDefault()

  // 鼠标是否按下的标识
  isMouseDown = true

  // 获取鼠标按下时的点位信息
  const point = {
    x: e.offsetX,
    y: e.offsetY
  }

  // 绘制放大镜
  drawCavansWithManifier(true, point)
}

// 监听鼠标抬起的事件
canvas.onmouseup = function (e) {
  e.preventDefault()
  isMouseDown = false

  // 不绘制放大镜
  drawCavansWithManifier(false)
}

// 监听鼠标移出时的事件
canvas.onmouseout = function (e) {
  e.preventDefault()
  isMouseDown = false
  drawCavansWithManifier(false)
}

// 监听鼠标移动的事件
canvas.onmousemove = function (e) {
  e.preventDefault()

  // 如果鼠标按下的情况下
  if (isMouseDown) {
    const point = {
      x: e.offsetX,
      y: e.offsetY
    }
    drawCavansWithManifier(true, point)
  }
}

/**
 * 通过 Canvas 绘制放大镜
 * @param { Boolean } isShowMagnifier 是否显示放大镜
 * @param { Object } point 鼠标的点位信息
*/
function drawCavansWithManifier(isShowMagnifier, point) {
  // 这个方法通过把像素设置为透明以达到擦除一个矩形区域的目的。
  // 因为放大的时候，鼠标肯定会一直在移动，移动的过程中相当于在不断地绘制图形，
  // 所以需要不断的擦除画布，然后再绘制放大的图像
  ctx.clearRect(0, 0, canvas.width, canvas.height)

  ctx.drawImage(image, 0, 0, canvas.width, canvas.height)

  // 如果显示才绘制放大镜
  if (isShowMagnifier) drawManifier(point)
}

/**
 * 绘制放大镜
 * @param { Object } point 鼠标的点位信息
*/
function drawManifier(point) {
  // 显示放大镜内容的区域大小的半径
  const mr = 200

  // 因为大图和原图是按照 scale 缩放，所以需要把鼠标的点位信息也按照 scale 进行缩放
  // 拿到放大图对应原图中鼠标按下的点位
  const imageLG_cx = point.x * scale
  const imageLG_cy = point.y * scale

  // 当我们鼠标按下的时候拿到的位置信息是作为放大镜的中心点，
  // 所以在放大的图中，需要把中心点减去放大镜的半径，
  // 才能得到放大图的内容的左上角位置
  const sx = imageLG_cx - mr
  const sy = imageLG_cy - mr

  // 要把放大镜放到原图的哪个位置
  // 这里是把鼠标点下的位置作为放大镜的中心
  // 所以需要减去半径，得到放大镜左上角的位置
  const dx = point.x - mr
  const dy = point.y - mr

  ctx.save()

  ctx.linWith = 10.0
  ctx.strokeStyle = '#ccc'

  // 绘制放大镜
  ctx.beginPath()
  ctx.arc(point.x, point.y, mr, 0, Math.PI * 2, false) // 绘制一个圆作为放大镜的边框
  ctx.stroke() // 绘制
  ctx.clip() // 裁剪这个圆，后续绘制的内容都会在裁剪的区域内

  // 在原图中，对放大图进行裁剪并放到原图的指定位置
  ctx.drawImage(offCanvas, sx, sy, mr * 2, mr * 2, dx, dy, mr * 2, mr * 2)
  ctx.restore()
}
