const canvas = document.querySelector('canvas')
const ctx = canvas.getContext('2d') // 创建二维渲染上下文（相当于画笔）
const resetBtn = document.querySelector('.reset')
const submitBtn = document.querySelector('.submit')

ctx.lineWidth = 4 // 设置画笔宽度
ctx.lineCap = 'round' // 设置线条末端样式
ctx.lineJoin = 'round' // 设置线条连接处样式

// canvas 鼠标按下事件
canvas.onmousedown = e => {
  ctx.beginPath() // 新建一条路径

  // 将画笔起始点设置为当前点击的位置
  ctx.moveTo(e.offsetX, e.offsetY)

  // 鼠标按下后要开始绘制，所以在鼠标按下事件内监听鼠标移动事件
  canvas.onmousemove = ev => {
    // 当鼠标移动时，将画笔移动到当前鼠标位置，并绘制一条直线
    ctx.lineTo(ev.offsetX, ev.offsetY)
    ctx.stroke() // 绘制
  }
}

// canvas 鼠标抬起事件
canvas.onmouseup = e => {
  // 鼠标抬起后，我们就不需要监听鼠标移动事件了
  canvas.onmousemove = null
}

resetBtn.onclick = () => {
  // clearRect(x, y, width, height) 清除指定矩形区域，然后清空画布
  ctx.clearRect(0, 0, canvas.width, canvas.height)
}

// 点击提交按钮后，将画布内容转为图片
submitBtn.onclick = () => {
  const url = canvas.toDataURL()
  console.log(url);
  const file = dataURLToFile(url, 'sign.png')
  console.log(file);
  const fileUrl = URL.createObjectURL(file)
  console.log(fileUrl);
}

// base64 转为文件
const dataURLToFile = (dataURL, filename) => {
  const arr = dataURL.split(',')
  // 获取图片格式
  const imgType = arr[0].match(/:(.*?);/)[1]

  // atob() 方法用于解码使用 base-64 编码的字符串
  const dec = atob(arr[1])
  let n = dec.length
  const u8arr = new Uint8Array(n)

  while (n--) {
    // 转成ASCII码
    u8arr[n] = dec.charCodeAt(n)
  }
  return new File([u8arr], filename, { type: imgType })
}
