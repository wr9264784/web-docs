useImperativeHandle

useImperativeHandle 是 React 中的一个 Hook，它能让你自定义由 ref 暴露出来的句柄。

通常情况下，React 是 单向数据流，也就是由父组件把一些变量或者方法传递给子组件，但有时候需要在父组件中获取子组件的一些变量或方法，这时候用 useImperativeHandle 就是一个比较方便实现的方法，useImperativeHandle 的使用仍然需要借助 forwardRef。

useImperativeHandle(ref, createHandle, dependencies?)

import { useRef, forwardRef, useImperativeHandle } from "react";

// 子组件
// 通过 forward 包裹子组件后，子组件的参数中就会多一个 ref
const Children = (props: any, ref: any) => {
  const childrenFun = (): void => {
    console.log('这是子组件内部的方法');
  }

  const childMsg = '这是子组件内部的数据'

  // 在子组件中定义需要暴露给父组件的方法或数据
  useImperativeHandle(ref, () => {
    return {
      childMsg,
      childrenFun, // 这个函数可以在父组件中直接被调用
    }
  })

  return (
    <div>
      <h2>Children</h2>
    </div>
  )
}

// 父组件
const Parent = () => {
  // 通过 forward 包裹子组件后，子组件的参数中就会多一个 ref
  // 包裹后返回的仍然是一个组件，只不过该组件允许使用 ref 将 DOM 节点暴露给父组件。
  const Com = forwardRef(Children)
  const comRef = useRef(null)

  const showChildrenData = () => {
    if (comRef && comRef.current) {
      const { childMsg, childrenFun } = comRef.current
      console.log(childMsg) // 获取子组件的变量

      // 直接调用会报 ts 错误：此表达式不可调用。类型 "never" 没有调用签名
      // @ts-ignore
      childrenFun()
    }
  }

  return (
    <div>
      <div style={{ marginBottom: '12px' }}>
        <h1>Parent</h1>
        <button style={{ padding: '6px' }} onClick={ () => showChildrenData() }>点击按钮，调用子组件的变量和方法</button>
      </div>
      <Com ref={ comRef } />
    </div>
  )
}

export default Home
