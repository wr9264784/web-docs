import { createSlice } from "@reduxjs/toolkit";
import axios from 'axios'

import type { AppDispatch } from "../type";
import { stateType } from './takeaway.types'

// 定义初始化状态，并设置对应类型
const initialState: stateType = {
  foodsList: [], // 商品列表
}

const foodsStore = createSlice({
  // 模块名称，名称在项目中是独一无二的
  name: 'foods',

  // 初始化的状态
  initialState,

  // 同步方法，支持直接修改
  reducers: {
    // 设置商品列表
    setFoodsList(state, action) {
      state.foodsList = action.payload
    }
  }
})

// 获取 reducer 函数
const foodStoreReducer = foodsStore.reducer

// 获取 action
const { setFoodsList } = foodsStore.actions

// 以默认导出的方式导出 store 中的 reducer
export default foodStoreReducer

/**
 * 异步获取商品列表的方法
 */
export function fetFoodsList() {
  return async function(dispatch: AppDispatch) {
    // 接口请求获取数据
    const res = await axios.get('http://localhost:3004/takeaway')
    
    // 调用 dispatch 函数，提交一个 action
    dispatch(setFoodsList(res.data))
  }
}

5. 商品列表切换显示

如上图所示：需求是根据点击左侧不同的菜单，显示右侧不同的商品，在 /src/App.tsx 中，通过观察 <FoodsCategory /> 组件可以发现，我们是通过 foodsList.map 渲染了所有 tab 菜单栏 的 商品列表，所以，我们可以通过 activeIndex 是否等于当前遍历的列表项的 索引 来控制 当前菜单栏 对应的 商品列表 是否显示：

const App = () => {
  // 使用商品列表数据
  const { foodsList, activeIndex } = useSelector((state: RootState) => state.foods)

7. 统计区域实现

实现思路：

  1. 基于 store 中的 cartList 中所有商品列表项的 count 字段的值相加作为渲染数量。
  2. 基于 store 中的 cartList 累加 price * count。
  3. 购物车 cartList 的 length 不为零则高亮。

在 /src/components/Car/index.tsx 中，实现对应逻辑：

import classNames from 'classnames'
import Count from '../Count'
import { useSelector } from 'react-redux'
import type { RootState } from '../../store/type'
import './index.scss'

const Cart = () => {
  const { cartList } = useSelector((state: RootState) => state.foods)

  // 计算商品总量
  const count = cartList.reduce((accumulator, currentValue) => {
    if (!currentValue.count) currentValue.count = 0
    return accumulator + currentValue.count
  }, 0,)

  // 计算商品总价
  const sum = cartList.reduce((accumulator, currentValue) => {
    if (!currentValue.count) currentValue.count = 0
    return accumulator + currentValue.count * currentValue.price
  }, 0,)

  return (
    <div className="cartContainer">
      {/* 遮罩层 添加visible类名可以显示出来 */}
      <div className={classNames('cartOverlay')}></div>
      
      <div className="cart">
        {/* fill 添加fill类名可以切换购物车状态*/}
        {/* 购物车数量 */}
        <div className={classNames('icon', count > 0 && 'fill')}>
          {
            count > 0 && 
            <div className="cartCornerMark">{ count }</div>
          }
        </div>

        {/* 购物车价格 */}
        <div className="main">
          <div className="price">
            <span className="payableAmount">
              <span className="payableAmountUnit">¥</span>
              { sum }
            </span>
          </div>
          <span className="text">预估另需配送费 ¥5</span>
        </div>

        {/* 结算 or 起送 */}
        {
          sum >= 20
          ?
          <div className="goToPreview">去结算</div>
          :
          <div className="minFee">¥20起送</div>
        }
      </div>
      
      {/* 添加visible类名 div会显示出来 */}
      <div className={classNames('cartPanel')}>
        <div className="header">
          <span className="text">购物车</span>
          <span className="clearCart">
            清空购物车
          </span>
        </div>

        {/* 购物车列表 */}
        <div className="scrollArea">
          {cartList.map((item: any) => {
            return (
              <div className="cartItem" key={item.id}>
                <img className="shopPic" src={item.picture} alt="" />
                <div className="main">
                  <div className="skuInfo">
                    <div className="name">{item.name}</div>
                  </div>
                  <div className="payableAmount">
                    <span className="yuan">¥</span>
                    <span className="price">{item.price}</span>
                  </div>
                </div>
                <div className="skuBtnWrapper btnGroup">
                  <Count
                    count={item.count}
                  />
                </div>
              </div>
            )
          })}
        </div>
      </div>
    </div>
  )
}

export default Cart

8. 购物车列表功能实现

购物车列表功能需求如下图所示：

实现步骤如下图所示：

  1. 在 /src/store/modules/takeaway.ts 中，添加 新增商品数量 和 减少商品数量 以及 清空购物车列表 的 action：
import { createSlice } from "@reduxjs/toolkit";
import axios from 'axios'

import type { AppDispatch } from "../type";
import { stateType } from './takeaway.types'

// 定义初始化状态，并设置对应类型
const initialState: stateType = {
  foodsList: [], // 商品列表
  activeIndex: 0, // 默认选中的菜单索引值
  cartList: [], // 购物车列表
}

const foodsStore = createSlice({
  // 模块名称，名称在项目中是独一无二的
  name: 'foods',

  // 初始化的状态
  initialState,

  // 同步方法，支持直接修改
  reducers: {
    // 设置商品列表
    setFoodsList(state, action) {
      state.foodsList = action.payload
    },

    // 设置激活的菜单索引值
    setActiveIndex(state, action) {
      state.activeIndex = action.payload
    },

    // 添加到购物车列表
    addToCartList(state, action) {
      // 获取购物车中是否有当前添加的商品项
      const item = state.cartList.find(f => f.id === action.payload.id)
      
      // 如果添加过
      if (item) (item.count as number)++
      
      // 如果没有添加过，直接添加到购物车列表
      else {
        action.payload.count = 1
        state.cartList.push(action.payload)
      }
    },

    // 增加商品的数量
    inCount(state, action) {
      const item = state.cartList.find(item => item.id === action.payload.id)
      if (item && item.count) {
        item.count++
      }
    },

    // 减少商品的数量
    deCount(state, action) {
      // 如果数量 大于等于 2 的情况下，继续减一
      const item = state.cartList.find(item => item.id === action.payload.id)
      if (item && item.count && item.count >= 2) {
        item.count--
      }
      // 如果小于 2 了，也就是当前数量是 1，再减就是 0 了，需要在列表去掉当前商品
      else {
        const index = state.cartList.findIndex(f => f.id === action.payload.id)
        state.cartList.splice(index, 1)
      }
    },

    // 清空购物车
    clearCartList(state) {
      state.cartList = []
    }
  }
})

// 获取 reducer 函数
const foodStoreReducer = foodsStore.reducer

// 获取 action
const { setFoodsList, setActiveIndex, addToCartList, inCount, deCount, clearCartList } = foodsStore.actions

// 以默认导出的方式导出 store 中的 reducer
export default foodStoreReducer

// 导出 action
export { setActiveIndex, addToCartList, inCount, deCount, clearCartList }

  }
}
