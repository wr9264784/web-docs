生成 Token

安装配置 jwt

  1. 安装 egg-jwt 插件：npm i egg-jwt --save

  2. 在 config/plugin.js 中引入插件：

jwt: {
  enable: true,
  package: 'egg-jwt'
}
  1. 配置 config/config.default.js 文件，设置 secret，注意 secret 不能泄露。
config.jwt = {
  secret: 'this_is_secret' // 加密密钥，可以更换为更复杂的字符串
}

配置路由

/api/lgoin 接口对应的 controller：

const { Controller } = require('egg');

class LoginController extends Controller {
  async login() {
    // token 就是一个加密的字符串，那么用什么来生成 token 呢？
    // 其实是用一个 js 对象来生成的。
    const user = {
      username: 'admin',
      password: '123456',
    };

    const term = (60 * 60 * 24) + 's'; // token有效期，设置为24小时

    // 当我们配置了 jwt 后，就可以访问 this.app.jwt
    // 这里是通过 jwt 的 sign 方法来生成 token 的，
    // 第一个参数是加密的对象，第二个参数是加密密钥。
    const token = this.app.jwt.sign(user, this.app.config.jwt.secret, { expiresIn: term });

    this.ctx.body = {
      code: 200,
      data: token,
    };
  }
}

module.exports = LoginController;

/api/verifyToken 接口对应的 controller，用来校验 token：


const { Controller } = require('egg');

class VerifyTokenController extends Controller {
  async verifyToken() {
    const token = this.ctx.get('token');

    // 校验 token
    try {
      const decode = this.app.jwt.verify(token, this.app.config.jwt.secret);

      this.ctx.body = {
        code: 200,
        data: decode,
      };
    } catch (error) {
      this.ctx.body = {
        code: -9999,
        msg: 'token 校验失败',
      };
    }

  }
}

module.exports = VerifyTokenController;
